	void visitAndExtract(ASTNode node,int level,Pair p) {
		String[] tokens= getTokens(node);
		
		for t in tokens{
		    String sToken=getPrefix(t,level);
		    p.getSource().append(sToken);
		    p.getTarget().append(t);
        }
	}
	...
}
